#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() {

    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    char a[100],b[100];int i,j,k=0;char c[100],d[100];
    scanf("%s%s",&a,&b);
    j=0;
    for(i=0;i<strlen(a);i++)
    {
        for(;j<strlen(a);j++)
        {
            if(a[i]==b[j])
            {
                c[k]=a[i];
                ++k;
                break;
            }
        }
    }
    k=0;j=0;
    for(i=0;i<strlen(a);i++)
    {
        for(;j<strlen(a);j++)
        {
            if(a[j]==b[i])
            {
                d[k]=a[j];
                ++k;
                break;
            }
        }
    }
    if(strlen(c)>=strlen(d))
    {
        for(i=0;i<strlen(c);i++)
            printf("%c",c[i]);
    }
    else
    {
        for(i=0;i<strlen(d);i++)
            printf("%c",d[i]);
    }
    return 0;
}
