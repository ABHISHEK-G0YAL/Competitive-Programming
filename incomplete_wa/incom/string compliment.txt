#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main()
{
    int n,i;
    scanf("%d",&n);
    char a[n][50];
    for(i=0;i<n;i++)
    scanf("%s",a[n]);
    for(i=0;i<n;i++)
    {
        for(i=0;i<strlen(a[n]);i++)
        {
            if(a[n][i]>=65 && a[n][i]<=90)
            a[n][i]=90-(a[n][i]-65);
            else if(a[n][i]>=97 && a[n][i]<=122)
            a[n][i]=122-(a[n][i]-97);
        }
        printf("%s\n",a[n]);
    }
    return 0;
}
